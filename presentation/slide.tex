		\item Somtimes remote debugging with IDE (Eclipse) is very slow
		\item Existing debugger \emph{jdb} has some imperfections
	\end{itemize}
\end{frame}

\section{Problems}
\begin{frame}{Problems with \emph{jdb}}	
	\begin{itemize}
		\item Very long commands arguments - no autocompletion
		\item Using same commands - no command history
		\item No condition breakpoints
		\item Conslusion: \emph{jdb} shell has no advanced features
	\end{itemize}
\end{frame}
